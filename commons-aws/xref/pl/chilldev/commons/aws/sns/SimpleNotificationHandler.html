8   package pl.chilldev.commons.aws.sns;
9   
10  import java.util.function.Consumer;
11  
12  import com.amazonaws.services.lambda.runtime.events.SNSEvent;
13  
14  /**
15   * SNS notifications handler that simply processes notification body.
16   */
17  public class SimpleNotificationHandler extends NotificationHandler
18  {
19      /**
20       * Initializes notification handler.
21       *
22       * @param messageBodyHandler SNS message handler
23       */
24      public SimpleNotificationHandler(Consumer<String> messageBodyHandler)
25      {
26          super((SNSEvent.SNS data) -> messageBodyHandler.accept(data.getMessage()));
27      }
28  }
