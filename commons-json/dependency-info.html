Dependency Information

Apache Maven

<dependency>
  <groupId>pl.wrzasq.commons</groupId>
  <artifactId>commons-json</artifactId>
  <version>2.0.7</version>
</dependency>

Apache Buildr

'pl.wrzasq.commons:commons-json:jar:2.0.7'

Apache Ivy

<dependency org="pl.wrzasq.commons" name="commons-json" rev="2.0.7">
  <artifact name="commons-json" type="jar" />
</dependency>

Groovy Grape

@Grapes(
@Grab(group='pl.wrzasq.commons', module='commons-json', version='2.0.7')
)

Gradle/Grails

compile 'pl.wrzasq.commons:commons-json:2.0.7'

Scala SBT

libraryDependencies += "pl.wrzasq.commons" % "commons-json" % "2.0.7"

Leiningen
