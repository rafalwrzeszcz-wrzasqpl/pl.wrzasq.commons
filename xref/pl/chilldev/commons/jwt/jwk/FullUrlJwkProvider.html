8   package pl.chilldev.commons.jwt.jwk;
9   
10  import java.net.MalformedURLException;
11  import java.net.URL;
12  
13  import com.auth0.jwk.UrlJwkProvider;
14  
15  /**
16   * Wrapper for URL-based JWK provider that does support keys in deeper locations.
17   */
18  public class FullUrlJwkProvider extends UrlJwkProvider
19  {
20      /**
21       * Initializes JWK provider for given issue URL.
22       *
23       * @param issuer Issuer URL.
24       */
25      public FullUrlJwkProvider(String issuer)
26      {
27          super(FullUrlJwkProvider.urlForIssuer(issuer));
28      }
29  
30      /**
31       * Builds URL to JWKS file.
32       *
33       * @param issuer Issuer URL.
34       * @return JWKS file URL.
35       */
36      private static URL urlForIssuer(String issuer)
37      {
38          try {
39              return new URL(issuer + "/.well-known/jwks.json");
40          } catch (MalformedURLException error) {
41              throw new IllegalArgumentException("Invalid JWKS URI.", error);
42          }
43      }
44  }
