8   package pl.chilldev.commons.text.formatter;
9   
10  import org.pegdown.Extensions;
11  import org.pegdown.PegDownProcessor;
12  
13  /**
14   * Markdown format handler.
15   */
16  public class MarkdownFormatter implements FormatterInterface
17  {
18      /**
19       * Markdown processor with all available extras.
20       */
21      private PegDownProcessor markdownProcessor = new PegDownProcessor(Extensions.ALL);
22  
23      /**
24       * {@inheritDoc}
25       */
26      @Override
27      public String transform(String text)
28      {
29          return this.markdownProcessor.markdownToHtml(text);
30      }
31  }
