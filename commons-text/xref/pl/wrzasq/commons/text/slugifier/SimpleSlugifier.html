10  import java.text.Normalizer;
11  import java.util.Locale;
12  
13  /**
14   * Simple implementation of slugifier.
15   */
16  public class SimpleSlugifier implements Slugifier {
17      /**
18       * Default separator.
19       */
20      public static final String DEFAULT_DELIMITER = "-";
21      /**
22       * Separator for URL parts.
23       */
24      private String delimiter = SimpleSlugifier.DEFAULT_DELIMITER;
25  
26      /**
27       * {@inheritDoc}
28       */
29      @Override
30      public String slugify(String text) {
31          return Normalizer.normalize(text, Normalizer.Form.NFD).replaceAll("[^\\p{ASCII}]", "").replaceAll("\\W+", this.delimiter).replaceAll(this.delimiter + "+", this.delimiter).replaceAll("^" + this.delimiter, "").replaceAll(this.delimiter + "$", "").toLowerCase(Locale.ROOT);
32      }
33  
34      /**
35       * {@inheritDoc}
36       */
37      @Override
38      public String slugify(String... texts) {
39          return this.slugify(String.join(this.delimiter, texts));
40      }
41

43       * Separator for URL parts.
44       */
45      @SuppressWarnings("all")
